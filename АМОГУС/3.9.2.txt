#include <iostream>
#include <unordered_map>
#include <vector>
using namespace std;

int main() {
    int n, m;
    cin >> n;
    vector<int> K(n);
    unordered_map<int, int> index;

    for (int i = 0; i < n; i++) {
        cin >> K[i];
        index[K[i]] = i;
    }

    cin >> m;
    vector<int> qi(m);
    for (int i = 0; i < m; i++) {
        cin >> qi[i];
        if (index.find(qi[i]) != index.end()) {
            cout << index[qi[i]] << endl;
        } else {
            cout << -1 << endl;
        }
    }

    return 0;
}
