#include <iostream>
#include <vector>

void findMinCoinSet(int money) {
    std::vector<int> coins = {50, 20, 10, 5, 1};
    std::vector<int> result;

    for (int i = 0; i < coins.size(); ++i) {
        while (money >= coins[i]) {
            result.push_back(coins[i]);
            money -= coins[i];
        }
    }

    std::cout << result.size() << std::endl;
    for (int i = 0; i < result.size(); ++i) {
        std::cout << result[i];
        if (i < result.size() - 1) {
            std::cout << " ";
        }
    }
    std::cout << std::endl;
}

int main() {
    int money;
    std::cin >> money;
    findMinCoinSet(money);
    return 0;
}
