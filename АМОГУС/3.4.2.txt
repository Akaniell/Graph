#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    int n, s;
    std::cin >> n >> s;
    std::vector<int> prices(n);

    for (int i = 0; i < n; ++i) {
        std::cin >> prices[i];
    }

    std::sort(prices.begin(), prices.end());

    int total = 0;
    int count = 0;
    for (int i = 0; i < n; ++i) {
        total += prices[i];
        if (total <= s) {
            count++;
        } else {
            break;
        }
    }

    std::cout << count << std::endl;

    return 0;
}
