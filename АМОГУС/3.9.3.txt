#include <iostream>
#include <unordered_map>
#include <vector>
using namespace std;

int main() {
    int n, m;
    cin >> n;
    vector<int> K(n);
    unordered_map<int, int> count;

    for (int i = 0; i < n; i++) {
        cin >> K[i];
        count[K[i]]++;
    }

    cin >> m;
    vector<int> qi(m);
    for (int i = 0; i < m; i++) {
        cin >> qi[i];
        cout << count[qi[i]] << " ";
    }

    return 0;
}
